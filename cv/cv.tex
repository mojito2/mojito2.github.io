%  A triangulated heptagon — a vertex of the associahedron, the polytope at the
%  centre of Giulio's amplitudes work.
\newcommand{\heptagon}{%
  \begin{tikzpicture}[scale=0.40]
    \def\R{2.6}
    \foreach \i in {0,...,6}{\coordinate (v\i) at ({90+\i*360/7}:\R);}
    \filldraw[fill=wash,draw=clay,line width=1.1pt,line join=round]
      (v0)--(v1)--(v2)--(v3)--(v4)--(v5)--(v6)--cycle;
    \draw[clay,line width=0.7pt,opacity=0.6] (v0)--(v2) (v0)--(v3) (v0)--(v4) (v0)--(v5);
    \foreach \i in {0,...,6}{\fill[clay] (v\i) circle (1.7pt);}
  \end{tikzpicture}}
